
### **Como rodar:**
1. Salve em `pedidos_otimizado.py`
2. Execute: `python pedidos_otimizado.py`

### **Código completo testável:**
```python
import time

class Pedido:
    def __init__(self, nome_produto, preco, quantidade):
        self.nome_produto = nome_produto
        self.preco = preco
        self.quantidade = quantidade

    def get_subtotal(self):
        return self.preco * self.quantidade

# Versão ANTIGA (lenta)
def processador_pedidos_lento():
    lista_pedidos = []
    
    for i in range(100):  # Reduzido para 100 para não travar
        lista_pedidos.append(Pedido(f"Produto {i}", i + 10.0, 2))
    
    print("Processando com método LENTO...")
    inicio = time.time()
    
    for i in range(len(lista_pedidos)):
        total_geral = 0
        for j in range(len(lista_pedidos)):
            total_geral += lista_pedidos[j].get_subtotal()
        if i % 20 == 0:  # Mostra apenas alguns para não poluir
            print(f"Pedido {i}: Total geral: {total_geral}")
    
    fim = time.time()
    print(f"Tempo método lento: {fim - inicio:.4f} segundos\n")

# Versão NOVA (rápida)
def processador_pedidos_rapido():
    lista_pedidos = []
    
    for i in range(100):
        lista_pedidos.append(Pedido(f"Produto {i}", i + 10.0, 2))
    
    print("Processando com método RÁPIDO...")
    inicio = time.time()
    
    total_geral = sum(pedido.get_subtotal() for pedido in lista_pedidos)
    
    for i, pedido in enumerate(lista_pedidos):
        if i % 20 == 0:
            print(f"Pedido {i}: Total geral: {total_geral}")
    
    fim = time.time()
    print(f"Tempo método rápido: {fim - inicio:.4f} segundos")

# Executar comparação
if __name__ == "__main__":
    print("=== COMPARAÇÃO DE PERFORMANCE ===")
    processador_pedidos_lento()
    processador_pedidos_rapido()
```
